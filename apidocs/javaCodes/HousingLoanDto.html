public class HousingLoanDto implements Serializable{

    private String data; //日期，时间
    private double repayment; //还款金额
    private double repaymentPrincipal; //还款 本金
    private double repaymentInterest; //还款 利息
    private double repaymentRate; //还款年利率
    private int repaymentTime; //贷款期限  月

    @JSONField(name="data")
    public String getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(String data){
        this.data = data;
    }

    @JSONField(name="repayment")
    public double getRepayment(){
        return repayment;
    }

    @JSONField(name="repayment")
    public void setRepayment(double repayment){
        this.repayment = repayment;
    }

    @JSONField(name="repaymentPrincipal")
    public double getRepaymentPrincipal(){
        return repaymentPrincipal;
    }

    @JSONField(name="repaymentPrincipal")
    public void setRepaymentPrincipal(double repaymentPrincipal){
        this.repaymentPrincipal = repaymentPrincipal;
    }

    @JSONField(name="repaymentInterest")
    public double getRepaymentInterest(){
        return repaymentInterest;
    }

    @JSONField(name="repaymentInterest")
    public void setRepaymentInterest(double repaymentInterest){
        this.repaymentInterest = repaymentInterest;
    }

    @JSONField(name="repaymentRate")
    public double getRepaymentRate(){
        return repaymentRate;
    }

    @JSONField(name="repaymentRate")
    public void setRepaymentRate(double repaymentRate){
        this.repaymentRate = repaymentRate;
    }

    @JSONField(name="repaymentTime")
    public int getRepaymentTime(){
        return repaymentTime;
    }

    @JSONField(name="repaymentTime")
    public void setRepaymentTime(int repaymentTime){
        this.repaymentTime = repaymentTime;
    }

}
